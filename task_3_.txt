import numpy as np  #Библиотека для математических операций и работы с массивами

# Создаем матрицу 5x5 со случайными числами от 1 до 10
matrix = np.random.randint(1, 11, (5, 5)) # мин.1, макс.11 (не включается поэтому до 10), размер матрицы

# Вычисляем характеристики
print("Матрица:\n", matrix) # вывод матрицы, перенос строки перед выводом матрицы
print("\nСреднее:", np.mean(matrix)) # вычисление и вывод сред.знач, вычисляет среднее всех ариф.матрицы
print("Максимум:", np.max(matrix)) # находит и выводит максимальное, возвращает максимальное знач
print("Минимум:", np.min(matrix)) # вычисляет и вывод минимальное
print("Сумма столбцов:", np.sum(matrix, 0)) #вычисляет и вывод из столбцов, суммирует,сум.вдоль столбцов
