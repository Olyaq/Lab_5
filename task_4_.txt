import numpy as np # импорт библиотеки для работы с массивами и мат.операциями
import matplotlib.pyplot as plt # импорт модуля из библиотеки

# Создаем матрицу 5x5
matrix = np.random.randint(1, 11, (5, 5))

# Выводим матрицу
print("Матрица:\n", matrix)

# Строим тепловую карту
plt.imshow(matrix, cmap='viridis') #создаем, отображает матрицу как изображение, данные для показа, использование цветов
plt.colorbar() # создает шкалу использования цветов знач.матрицы

# Добавляем числа в ячейки
for i in range(5): # индекс строки
    for j in range(5): # индекс
        plt.text(j, i, matrix[i,j], #добавляем текст в ячейку, отображает значение
                 ha='center', va='center', # горизонтальное и вертикальное по центру
                 color='w' if matrix[i,j] > 5 else 'k') # цвет текста зависит от знач.

plt.show() # показывает график
