# Импорт необходимых библиотек
import numpy as np  # Библиотека для мат.операций и работы с массивами
import matplotlib.pyplot as plt  # Библиотека для построения графиков

# Генерация случайных данных 
data = np.random.normal(size=1000) # создаем массив из 1000 значений, среднее 0, стандартное откл. 1

# Построение гистограммы
plt.hist(data, bins=20) # фун.постороение, массив, кол-во столбцов  

# Настройки графика
plt.title('Гистограмма нормального распределения') # заголовок 
plt.xlabel('Значения') # подпись
plt.ylabel('Частота') 

plt.grid(True) # включение сетки
plt.show() # вывод Гистограммы
