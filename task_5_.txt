import numpy as np # Библиотека для математических вычислений
import matplotlib.pyplot as plt # Библиотека для построения графиков

# Создаем 3 графика в одном ряду
fig, ax = plt.subplots(1, 3, figsize=(12, 3)) # объект всего изображения, массив из 3 осей, создаем область, размер всего изображения

# 1. График линейной функции
x = np.linspace(0, 5, 30) # создаем массив x с 30 точками от 0 до 5
ax[0].plot(x, x**2, 'r') # строим график y = x^2, желтый линия
ax[0].set_title('y = x²') # устанавливаем заголовок график

# 2. Точечный график
x, y = np.random.rand(2, 20) # генерируем 2 массива по 20 случайных чисел от 0 до 1
ax[1].scatter(x, y, c='g') # рисуем точки, красный цвет
ax[1].set_title('Случайные точки') # устанавливаем заголовок

# 3. Столбчатая диаграмма
ax[2].bar(['A','B','C'], [3,7,2]) # рисуем столбцы с высотами 
ax[2].set_title('Категории') # устанавливаем заголовок

plt.tight_layout() # делаем отступы между графиками
plt.show() #выводим графики
